Тесты на POST запросы

Тест на добавление сотрудника

class EmployeeCreateTest(TestCase):
    def test_create_employee(self):
        url = reverse('candy_app:employee-create')
        data = {'id': 1, 'name': 'Харитонов Лука Ильич', 'phone_number': '+79949788813',
                'passport'= 4952 931368}
        response = self.client.post(url, data, format='json')
        self.assertEqual(response.status_code, status.HTTP_201_CREATED)
        self.assertEqual(response.data, data)

Тест на поручение заказа

class RequestCreateTest(TestCase):
    @classmethod
    def setUpTestData(cls):
        Candy.objects.create(name='Ferrero Rocher', sort='Конфеты', price='400')

    def test_create_request(self):
        url = reverse('candy_app:request-create')
        data = {"id": 1, "candy": 1, "name": 'Харитонов Лука Ильич'}
        response = self.client.post(url, data, format='json')
        self.assertEqual(response.status_code, status.HTTP_201_CREATED)
        self.assertEqual(response.data, data)
